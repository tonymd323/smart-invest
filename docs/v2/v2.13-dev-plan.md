## Alpine.js 使用模式

```html
<!-- 加载 -->
<script src="/static/alpine.min.js" defer></script>

<!-- 模态框 -->
<div x-data="{ open: false }">
  <button @click="open = true">卖出</button>
  <div x-show="open" x-transition>
    <p>确认卖出？</p>
    <button @click="submitSell()">确认</button>
    <button @click="open = false">取消</button>
  </div>
</div>

<!-- 折叠区 -->
<div x-data="{ expanded: false }">
  <button @click="expanded = !expanded">
    <span x-text="expanded ? '收起' : '展开'"></span>
  </button>
  <div x-show="expanded" x-collapse>内容</div>
</div>

<!-- Loading 状态 -->
<div x-data="{ loading: false }">
  <button @click="loading = true; fetch(...).finally(() => loading = false)"
          :disabled="loading">
    <span x-show="!loading">提交</span>
    <span x-show="loading">处理中...</span>
  </button>
</div>
```
